```ts
type AccountState<TAccount> =
  | {
      account: Promise<SupportedAccount<TAccount>>;
      error?: never;
      status: "INITIALIZING";
    }
  | {
      account: {
        address: Address;
      };
      error?: never;
      status: "RECONNECTING";
    }
  | {
      account: SupportedAccount<TAccount>;
      error?: never;
      status: "READY";
    }
  | {
      account: undefined;
      error?: never;
      status: "DISCONNECTED";
    }
  | {
      account: undefined;
      error: Error;
      status: "ERROR";
    };
```
