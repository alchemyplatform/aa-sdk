```ts
function prepareSign<TAccount>(
  client,
  params,
): Promise<{
  chainId: `0x${string}`;
  signatureRequest:
    | {
        data:
          | string
          | {
              raw: `0x${string}`;
            };
        type: "personal_sign";
      }
    | {
        data: {
          domain?: {
            chainId?: number;
            name?: string;
            salt?: `0x${string}`;
            verifyingContract?: `0x${string}`;
            version?: string;
          };
          message: {
            [key: string]: unknown;
          };
          primaryType: string;
          types: {
            [key: string]: object[];
          };
        };
        type: "eth_signTypedData_v4";
      };
}>;
```

Defined in: [account-kit/wallet-client/src/client/actions/prepareSign.ts:37](https://github.com/alchemyplatform/aa-sdk/blob/da19afd63b0ba06a4de95b0d2d7871f64d546304/account-kit/wallet-client/src/client/actions/prepareSign.ts#L37)

Prepares a signature request for signing messages or transactions.

## Example

```ts
// Prepare a message to be signed
const result = await client.prepareSign({
  from: "0x1234...",
  type: "personal_sign",
  data: "Hello, world!",
});
```
