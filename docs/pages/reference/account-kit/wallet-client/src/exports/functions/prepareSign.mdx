  client,
  params,
): Promise<{
  chainId: `0x${string}`;
  signatureRequest:
    | {
        data:

          };
          primaryType: string;
          types: {
            [key: string]: object[];
          };
        };
        type: "eth_signTypedData_v4";
      };
}>;

Defined in: [account-kit/wallet-client/src/client/actions/prepareSign.ts:37](https://github.com/alchemyplatform/aa-sdk/blob/49e74eee8606f480b200b4612d00d950da8a8742/account-kit/wallet-client/src/client/actions/prepareSign.ts#L37)

Prepares a signature request for signing messages or transactions.

## Example

```ts
// Prepare a message to be signed
const result = await client.prepareSign({
  from: "0x1234...",
  type: "personal_sign",
  data: "Hello, world!",
});
```
