In React, you can use [Viem](https://viem.sh/docs/contract/encodeFunctionData) to encode function call data.

**Prerequisites**

- [Smart Wallets installed and configured in your project](/docs/wallets/pages/react/setup.mdx).

<CodeBlocks>

```tsx twoslash title="sendCalls.tsx" focus={20-24}
import { useSendCalls, useSmartAccountClient } from "@account-kit/react";
import { encodeFunctionData } from "viem";
import { exampleAbi } from "./abi.ts";

export default function SendCalls() {
  const { client } = useSmartAccountClient({});
  const { sendCalls, isSendingCalls } = useSendCalls({
    client,
  });

  const handleSend = async () => {
    if (!client) {
      throw new Error("Smart account client not connected");
    }

    sendCalls({
      calls: [
        {
          to: "0x0000000000000000000000000000000000000000",
          data: encodeFunctionData({
            abi: exampleAbi,
            functionName: "mintTo",
            args: [client.account.address],
          }),
        },
      ],
    });
  };

  return (
    <button onClick={handleSend}>
      {isSendingCalls ? "Sending..." : "Send"}
    </button>
  );
}
```

```ts twoslash title="abi.ts"
export const exampleAbi = [
  {
    inputs: [{ name: "recipient", type: "address" }],
    name: "mintTo",
    outputs: [{ name: "", type: "uint256" }],
    stateMutability: "nonpayable",
    type: "function",
  },
] as const;
```
