import { chain, transport, policyId } from "./config";
import { LocalAccountSigner } from "@aa-sdk/core";
import { generatePrivateKey } from "viem/accounts";

  transport,
  signer: LocalAccountSigner.privateKeyToAccountSigner(generatePrivateKey()),

});
